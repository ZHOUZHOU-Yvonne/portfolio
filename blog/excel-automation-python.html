场景一：多文件合并

财务每月发来30个部门的Excel，需要合并成一个总表：

import pandas as pd
from pathlib import Path

def merge_excel_files(folder, output):
    files = Path(folder).glob('*.xlsx')
    dfs = [pd.read_excel(f) for f in files]
    result = pd.concat(dfs, ignore_index=True)
    result.to_excel(output, index=False)
    print(f'Merged {len(dfs)} files → {output}')

merge_excel_files('部门报表/', '月度汇总.xlsx')

场景三：数据清洗

def clean_sales_data(df):
    df['金额'] = df['金额'].str.replace('[¥,]', '', regex=True)
    df['金额'] = pd.to_numeric(df['金额'], errors='coerce')
    df['日期'] = pd.to_datetime(df['日期'], errors='coerce')
    df = df.drop_duplicates(subset=['订单号'])
    return df.dropna(subset=['订单号', '金额'])
